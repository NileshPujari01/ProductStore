create table if not exists product_category(
	product_category_id int not null generated by default as identity,
	constraint pk_product_category_id primary key (product_category_id),
	product_category_name varchar(50)
);
cluster product_category using pk_product_category_id;


create table if not exists products(
	product_id int not null generated by default as identity,
	constraint pk_product_id primary key (product_id),
	product_name varchar(50),
	product_category int not null,
	constraint fk_product_category foreign key (product_category) references product_category(product_category_id),
	product_price numeric(5,2),
	product_rating int
);
cluster products using pk_product_id;

create table if not exists product_images(
	product_image_id int not null generated by default as identity,
	constraint pk_product_image_id primary key (product_image_id),
	product_id int not null,
	constraint fk_product_id foreign key (product_id) references products(product_id),
	image_name varchar(50)
);
cluster product_images using pk_product_image_id;
